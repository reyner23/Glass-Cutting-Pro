function rotateAllPieces(){ var rows=[...document.querySelectorAll('#cut-list > div[style]')]; if(!rows.length){toast('Tidak ada piece untuk di-rotate','warn');return} rows.forEach(function(r){ var wEl=r.querySelector('.cp-w'); var hEl=r.querySelector('.cp-h'); if(!wEl||!hEl)return; var w=wEl.value, h=hEl.value; wEl.value=h; hEl.value=w; }); toast('↻ Semua pieces di-rotate (W↔H ditukar)','info'); }
